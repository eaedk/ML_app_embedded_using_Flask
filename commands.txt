# Setup virtual environment

## Create the venv
python3 -m venv venv

## Activate the env
source venv/bin/activate        #this line will work in linux
venv\Scripts\activate           # this it the code  for windows.

## Deactivate the env
deactivate


# Install required packages
pip install --upgrade pip               # update pip first before to use it, to avoid warning
pip install -qr requirements.txt

# Config quick excecution
export FLASK_APP=script.py   #this line will work in linux
set FLASK_APP=script.py      # this it the code  for windows.
run flask
